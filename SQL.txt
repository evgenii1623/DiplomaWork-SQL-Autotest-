задание 1

SELECT cour.login,
       COUNT(ord.courierId) AS "delivery_Count"                              
FROM "Couriers" AS cour
INNER JOIN “Orders” AS ord ON cour.id = ord. "courierId"                
WHERE ord. "inDelivery" = true                                                               
GROUP BY cour_login;
